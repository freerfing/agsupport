```
---短信-接收用户映射表
CREATE TABLE AWATER.A_SMS_USER
   (	
    "SMS_ID" VARCHAR2(36 CHAR) NOT NULL ENABLE, 
	"RECEIVER_ID" VARCHAR2(36 CHAR), 
	"PHONE" VARCHAR2(20 CHAR), 
	"STATUS" VARCHAR2(1 CHAR) DEFAULT '0', 
	"ERROR_MSG" VARCHAR2(100 CHAR)
   );
   
   COMMENT ON TABLE A_SMS_USER IS '短信-接收人映射表';
   COMMENT ON COLUMN A_SMS_USER.SMS_ID IS '短信id';
   COMMENT ON COLUMN A_SMS_USER.RECEIVER_ID IS '短信内容';
   COMMENT ON COLUMN A_SMS_USER.PHONE IS '接受用户手机号';
   COMMENT ON COLUMN A_SMS_USER.STATUS IS '接收状态：0-保存，1-接收成功，2-接收失败，3-用户没有手机号';
   COMMENT ON COLUMN A_SMS_USER.ERROR_MSG IS '错误信息';

```
