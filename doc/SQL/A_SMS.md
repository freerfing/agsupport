```
---短信发送记录表
CREATE TABLE AWATER.A_SMS 
   (	
    ID VARCHAR2(36 CHAR) NOT NULL ENABLE, 
	CONTENT VARCHAR2(1000 CHAR), 
	TYPE VARCHAR2(1 CHAR) DEFAULT '0', 
	CREATE_TIME DATE NOT NULL ENABLE, 
	SEND_TIME DATE, 
	SENDER_ID VARCHAR2(36 CHAR) NOT NULL ENABLE, 
	RANDOM_CODE VARCHAR2(10 CHAR), 
	STATUS VARCHAR2(1 CHAR) DEFAULT '0'
	);
	
	
COMMENT ON TABLE A_SMS IS '短信发送记录表';
COMMENT ON COLUMN A_SMS.ID IS 'ID';
COMMENT ON COLUMN A_SMS.CONTENT IS '短信内容';
COMMENT ON COLUMN A_SMS.TYPE IS '类型信息，0：公告短信 1：其他';
COMMENT ON COLUMN A_SMS.CREATE_TIME IS '创建时间';
COMMENT ON COLUMN A_SMS.SEND_TIME IS '发送时间';
COMMENT ON COLUMN A_SMS.SENDER_ID IS '发送人id';
COMMENT ON COLUMN A_SMS.RANDOM_CODE IS '随机码';
COMMENT ON COLUMN A_SMS.STATUS IS '状态：0-保存，1-发送成功，2，全部失败，3-部分失败';

```
