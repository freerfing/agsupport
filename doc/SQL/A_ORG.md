--机构信息表
CREATE TABLE A_ORG (
    ID VARCHAR2(36 CHAR) NOT NULL,
    NAME VARCHAR2(256 CHAR),
    ORG_CODE VARCHAR2(36 CHAR),
    PARENT_ORG_CODE VARCHAR2(36 CHAR),
    XPATH VARCHAR2(200 CHAR),
    DISPORDER NUMBER(10) DEFAULT 9999999,
    CONSTRAINT PRI_A_ORG_ID PRIMARY KEY (ID)
);

COMMENT ON TABLE A_ORG IS '机构表';
COMMENT ON COLUMN A_ORG.ID IS 'ID';
COMMENT ON COLUMN A_ORG.NAME IS '名称';
COMMENT ON COLUMN A_ORG.ORG_CODE IS '机构编码';
COMMENT ON COLUMN A_ORG.PARENT_ORG_CODE IS '父机构编码';
COMMENT ON COLUMN A_ORG.XPATH IS '机构目录';
COMMENT ON COLUMN A_ORG.DISPORDER IS '排序';
