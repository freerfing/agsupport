CREATE TABLE A_USER(
	"ID" VARCHAR2(36 CHAR) NOT NULL ENABLE,  
	"LOGIN_NAME" VARCHAR2(50 CHAR), 
	"PASSWORD" VARCHAR2(100 CHAR),
	"USER_NAME" VARCHAR2(50 CHAR), 
	"IS_ACTIVE" VARCHAR2(1 CHAR), 
	"SEX" VARCHAR2(2 CHAR), 
	"TEL" VARCHAR2(20 CHAR), 
	"OFFICE_TEL" VARCHAR2(20 CHAR),
	"EMAIL" VARCHAR2(100 CHAR), 
	"CREATE_TIME" DATE, 
	"WORK_UNIT_NAME" VARCHAR2(256 CHAR),
	"WORK_UNIT_ADDRESS" VARCHAR2(256 CHAR), 
	"WORK_UNIT_CODE" VARCHAR2(50 CHAR), 
	"SELF_CERT" VARCHAR2(18 CHAR), 
	"JOB" VARCHAR2(50 CHAR),
	"FAX" VARCHAR2(15 CHAR), 
	"QQ" VARCHAR2(20 CHAR), 
);

COMMENT ON COLUMN A_USER.CREATE_TIME IS '创建时间';
