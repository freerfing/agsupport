--机构用户对应信息表
CREATE TABLE A_ORG_USER (
    ORG_ID VARCHAR2(36 CHAR),
    USER_ID VARCHAR2(36 CHAR),
    DISPORDER NUMBER NOT NULL
);

COMMENT ON TABLE A_ORG_USER IS '机构-用户表';
COMMENT ON COLUMN A_ORG_USER.ORG_ID IS '机构ID';
COMMENT ON COLUMN A_ORG_USER.USER_ID IS '用户ID';
COMMENT ON COLUMN A_ORG_USER.DISPORDER IS '用户排序';
