---元数据表信息表
CREATE TABLE A_METADATA_TABLE(
   ID VARCHAR2(40), 
   NAME VARCHAR2(40), 
   CNAME VARCHAR2(100), 
   DATABASEID VARCHAR2(100), 
   MODIFY_TIME DATE, 
   PRIID VARCHAR2(50), 
   DIRLAYERID VARCHAR2(50), 
   DISTFIELD VARCHAR2(255), 
   LAYER_CONFIG VARCHAR2(1000), 
   SEARCHFIELD VARCHAR2(500), 
   GROUPFIELD VARCHAR2(500), 
   SUMFIELD VARCHAR2(100), 
   FIELDUNIT VARCHAR2(500)
);

COMMENT ON TABLE A_METADATA_TABLE IS '元数据库信息表';
COMMENT ON COLUMN A_METADATA_TABLE.ID IS 'ID';
COMMENT ON COLUMN A_METADATA_TABLE.NAME IS '表英文名';
COMMENT ON COLUMN A_METADATA_TABLE.CNAME IS '表中文名';
COMMENT ON COLUMN A_METADATA_TABLE.DATABASEID IS '数据库ID';
COMMENT ON COLUMN A_METADATA_TABLE.MODIFY_TIME IS '更新时间';
COMMENT ON COLUMN A_METADATA_TABLE.PRIID IS '主键';
COMMENT ON COLUMN A_METADATA_TABLE.DIRLAYERID IS '图层标识';
COMMENT ON COLUMN A_METADATA_TABLE.DISTFIELD IS '地区字段';
COMMENT ON COLUMN A_METADATA_TABLE.LAYER_CONFIG IS '图层配置';
COMMENT ON COLUMN A_METADATA_TABLE.SEARCHFIELD IS '查询字段，以“,”隔开';
COMMENT ON COLUMN A_METADATA_TABLE.GROUPFIELD IS '分组字段';
COMMENT ON COLUMN A_METADATA_TABLE.SUMFIELD IS '统计累计字段',
COMMENT ON COLUMN A_METADATA_TABLE.FIELDUNIT IS '统计单位，条、个';

ALTER TABLE A_METADATA_TABLE ADD CONSTRAINT METADATA_TABLE_PK PRIMARY KEY(ID);
