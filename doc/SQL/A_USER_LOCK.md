-- 账号锁定表
CREATE TABLE A_USER_LOCK(
    ID VARCHAR2(36),
    LOGIN_NAME VARCHAR2(50),
    USER_NAME VARCHAR2(150),
	MOBILE VARCHAR2(20),
	QQ VARCHAR2(20),
	IS_LOCK VARCHAR2(1),
	LOCK_START_TIME DATE,
	LOCK_END_TIME DATE,
	ORG_PATH VARCHAR2(256),
	FAIL_LOGIN_TIMES NUMBER(2),
	PRIMARY KEY (ID)
);

COMMENT ON TABLE A_USER_LOCK IS '账号锁定表';
COMMENT ON COLUMN A_USER_LOCK.ID IS 'ID';
COMMENT ON COLUMN A_USER_LOCK.LOGIN_NAME IS '登录名';
COMMENT ON COLUMN A_USER_LOCK.USER_NAME IS '用户名';
COMMENT ON COLUMN A_USER_LOCK.MOBILE IS '手机号码';
COMMENT ON COLUMN A_USER_LOCK.QQ IS 'QQ号码';
COMMENT ON COLUMN A_USER_LOCK.IS_LOCK IS '是否锁定';
COMMENT ON COLUMN A_USER_LOCK.LOCK_START_TIME IS '锁定时间';
COMMENT ON COLUMN A_USER_LOCK.LOCK_END_TIME IS '解锁时间';
COMMENT ON COLUMN A_USER_LOCK.ORG_PATH IS '机构路径';
COMMENT ON COLUMN A_USER_LOCK.FAIL_LOGIN_TIMES IS '连续失败次数';
