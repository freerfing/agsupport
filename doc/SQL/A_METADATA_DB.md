CREATE TABLE A_METADATA_DB
(

    NAME VARCHAR2(40),

    URL VARCHAR2(100),
    CREATE_TIME DATE,
    MODIFY_TIME DATE
);

COMMENT ON COLUMN A_METADATA_DB.NAME IS '英文名';

COMMENT ON COLUMN A_METADATA_DB.URL IS 'url';
COMMENT ON COLUMN A_METADATA_DB.CREATE_TIME IS '创建时间';
COMMENT ON COLUMN A_METADATA_DB.MODIFY_TIME IS '最后更改时间';


ALTER TABLE A_METADATA_DB ADD CONSTRAINT METADATA_PK PRIMARY KEY(ID);
